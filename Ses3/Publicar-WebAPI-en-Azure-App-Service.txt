## INSTALACIÓN DE .NET 9 en WSL
11. sudo apt update
12. sudo apt install -y wget apt-transport-https software-properties-common
13. wget https://packages.microsoft.com/config/ubuntu/$(lsb_release -rs)/packages-microsoft-prod.deb -O packages-microsoft-prod.deb
14. sudo dpkg -i packages-microsoft-prod.deb
15. rm packages-microsoft-prod.deb
16. sudo apt update
17. sudo apt install -y dotnet-sdk-9.0

## PUBLICACIÓN
18. dotnet publish -c Release -o ./publish
19. cd publish
20. zip -r ../deploy.zip .  (los primeros puntos son la carpeta anterior
 y el último punto el directorio actual)
21. cd ..
22. az webapp deploy \
  --resource-group adventure-works-webapi_group \
  --name adventure-works-webapi \
  --src-path deploy.zip \
  --type zip
